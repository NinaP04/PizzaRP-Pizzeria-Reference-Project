# Hauptmenü
def hauptmenü():
    while True:
        print("\nBudget-Kategorien Menü")
        print("1. Kategorien anzeigen")
        print("2. Neue Kategorie hinzufügen")
        print("3. Kategorie bearbeiten")
        print("4. Kategorie löschen")
        print("5. Beenden")

        auswahl = input("Wähle eine Option (1-5): ")

        if auswahl == "1":
            anzeigen_kategorien()
        elif auswahl == "2":
            neue_kategorie_hinzufügen()
        elif auswahl == "3":
            kategorie_bearbeiten()
        elif auswahl == "4":
            kategorie_löschen()
        elif auswahl == "5":
            print("Programm beendet.")
            break
        else:
            print("Ungültige Eingabe. Bitte wähle eine Zahl zwischen 1 und 5.")

# Programm starten
hauptmenü()
